
git init : Start init
git add : Add files
git add -A : Add all files
git commit -m "message" : Commit changes
git status : Show status
git checkout : update or recover file
touch : create file
git checkout -f : Recover all files
git log : get commits
git log -p -number : commit of number
git rm --cached : remove file from repo
git status -s : short status
git branch -b : create/show branch
git remote add name path : add remote
git push name branch : push
git push -u : push in same branch
